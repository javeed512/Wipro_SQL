child table
===========



CREATE TABLE products_info (
  id INT,
  category_id INT,
  product_name VARCHAR(50),
  CONSTRAINT pk_product PRIMARY KEY (id),
  CONSTRAINT fk_prod_category
    FOREIGN KEY (category_id) REFERENCES category(id)
    ON DELETE  CASCADE/SET NULL
);
